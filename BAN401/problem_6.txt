CREATE TABLE CLIENTS (
	CLIENT_ID	INTEGER NOT NULL,
	CLIENT_NAME	TEXT,
	LOCATION	TEXT,
	CLIENT_CONTACT_INFO	TEXT,
	PRIMARY KEY(CLIENT_ID)
);
CREATE TABLE "BACKGROUND_SALES_AGENTS" (
	"STUDENT_ID"	INTEGER,
	"AGENT_ID"	INTEGER,
	"EDUCATION"	TEXT,
	"YEAR_OF_GRADUATION"	INTEGER,
	PRIMARY KEY("STUDENT_ID"),
	FOREIGN KEY("AGENT_ID") REFERENCES "SALES_AGENTS"("AGENT_ID")
);
CREATE TABLE CLIENT_ACCOUNT (
	CLIENT_ID	INTEGER NOT NULL,
	ACCOUNT_NUMBER	INTEGER,
	BALANCE	REAL,
	PRIMARY KEY(CLIENT_ID),
	FOREIGN KEY(CLIENT_ID) REFERENCES CLIENTS(CLIENT_ID)
); 
CREATE TABLE EXPORT_COMPANY (
	EXPORT_ID	INTEGER NOT NULL,
	EX_COMPANY_NAME	TEXT,
	LOCATION	TEXT,
	EX_CONTACT_INFO	TEXT,
	PRIMARY KEY(EXPORT_ID)
);
CREATE TABLE EXPORT_TRANSACTIONS (
	TRANSACTION_ID	INTEGER,
	DATE	TEXT,
	QUANTITY	INTEGER,
	TOTAL_VALUE	REAL,
	EXPORT_ID	INTEGER,
	SPECIES_ID	INTEGER,
	CLIENT_ID	INTEGER,
	PRIMARY KEY(TRANSACTION_ID),
	FOREIGN KEY(CLIENT_ID) REFERENCES CLIENTS(CLIENT_ID),
	FOREIGN KEY(EXPORT_ID) REFERENCES EXPORT_COMPANY(EXPORT_ID),
	FOREIGN KEY(SPECIES_ID) REFERENCES SPECIES(SPECIES_ID)
);
CREATE TABLE PRICE_HISTORY (
	PRICE_ID	INTEGER,
	DATE	TEXT,
	SPECIES_ID	INTEGER,
	PRICE_PR_KG	REAL,
	PRIMARY KEY(PRICE_ID),
	FOREIGN KEY(SPECIES_ID) REFERENCES SPECIES(SPECIES_ID)
);
CREATE TABLE SALES_AGENTS (
	AGENT_ID	INTEGER NOT NULL,
	SALES_AGENT_NAME	TEXT,
	EMAIL	TEXT,
	EXPORT_ID	INTEGER,
	PRIMARY KEY(AGENT_ID),
	FOREIGN KEY(EXPORT_ID) REFERENCES EXPORT_COMPANY(EXPORT_ID)
);
CREATE TABLE SEAFOOD_COMPANY (
	SF_COMPANY_NAME	TEXT,
	LOCATION	TEXT,
	SF_CONTACT_INFO	TEXT,
	COMPANY_ID	INTEGER NOT NULL,
	PRIMARY KEY(SF_COMPANY_NAME)
);
CREATE TABLE SEAFOOD_COMPANY_SPECIES (
	SF_COMPANY_NAME	TEXT,
	SPECIES_ID	INTEGER,
	PRIMARY KEY(SPECIES_ID,SF_COMPANY_NAME),
	FOREIGN KEY(SF_COMPANY_NAME) REFERENCES SEAFOOD_COMPANY(SF_COMPANY_NAME),
	FOREIGN KEY(SPECIES_ID) REFERENCES SPECIES(SPECIES_ID)
);
CREATE TABLE SPECIES (
	SPECIES_ID	INTEGER NOT NULL,
	SPECIES_NAME	TEXT,
	AVERAGE SIZE	REAL,
	PRIMARY KEY(SPECIES_ID)
);
CREATE TABLE STAFF (
	EMPLOYEE_ID	INTEGER NOT NULL,
	NAME 	TEXT,
	POSITION	TEXT,
	SALARY	REAL,
	SF_COMPANY_NAME	TEXT,
	PRIMARY KEY(EMPLOYEE_ID),
	FOREIGN KEY(SF_COMPANY_NAME) REFERENCES SEAFOOD_COMPANY(SF_COMPANY_NAME)
);
CREATE TABLE VESSELS (
	VESSEL_ID	INTEGER NOT NULL,
	NAME	 TEXT,
	SF_COMPANY_NAME	TEXT,
	CAPACITY	INTEGER,
	PRIMARY KEY(VESSEL_ID),
	FOREIGN KEY(SF_COMPANY_NAME) REFERENCES SEAFOOD_COMPANY(SF_COMPANY_NAME)
);
CREATE TABLE VESSEL_CATCHES (
	VESSEL_ID	INTEGER,
	SPECIES_ID	INTEGER,
	CATCH_QUANTITY 	INTEGER,
	PRIMARY KEY(VESSEL_ID,SPECIES_ID),
	FOREIGN KEY(SPECIES_ID) REFERENCES SPECIES(SPECIES_ID),
	FOREIGN KEY(VESSEL_ID) REFERENCES VESSELS(VESSEL_ID)
);
